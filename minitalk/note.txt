#include "minitalk.h"

void get_kill(int pid, char *str)
{
    unsigned int i;
    while (*str)
    {
        i = 8;
        while(i--)
        {
            if (*str >> i & 1)
                kill(pid, SIGUSR1);
            else
                kill(pid, SIGUSR2);
                usleep(50);
        }
        str++;
    }
}

int main (int argc, char **argv)
{

    if (argc != 3)
    {
        ft_printf("You must enter at least two arguments !\n");
        return (0);
    }
    get_kill(ft_atoi(argv[1]) ,argv[2]);
    return (0);
}
